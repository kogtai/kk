# MiniCraft Advanced Edition (Closer to Full Game) # Major upgrades: # - Proper collision system # - Camera scrolling # - Simple crafting # - Enemy mob (Zombie) # - Item drops # - Ender Dragon improved fight # # pip install pygame import pygame import random import math pygame.init() WIDTH, HEIGHT = 1000, 600 screen = pygame.display.set_mode((WIDTH, HEIGHT)) pygame.display.set_caption("MiniCraft Advanced") clock = pygame.time.Clock() TILE = 32 GRAVITY = 0.5 WORLD_WIDTH = 300 WORLD_HEIGHT = 120 COLORS = { "grass": (50, 200, 50), "dirt": (120, 72, 0), "stone": (100, 100, 100), "air": None } # Generate world world = [["air" for _ in range(WORLD_HEIGHT)] for _ in range(WORLD_WIDTH)] for x in range(WORLD_WIDTH): ground = random.randint(50, 70) for y in range(ground, WORLD_HEIGHT): if y == ground: world[x][y] = "grass" elif y < ground + 3: world[x][y] = "dirt" else: world[x][y] = "stone" # Player player = { "x": 200, "y": 0, "vx": 0, "vy": 0, "w": 28, "h": 48, "on_ground": False, "hp": 20, "inventory": {"dirt": 0, "stone": 0} } # Camera camera_x = 0 camera_y = 0 # Enemies zombies = [] # Items items = [] # Dragon dragon = {"x": 600, "y": 100, "hp": 300, "alive": False} in_end = False # Utilities def get_block(x, y): if 0 <= x < WORLD_WIDTH and 0 <= y < WORLD_HEIGHT: return world[x][y] return "stone" def solid(block): return block != "air" def rect_collide(px, py): for x in range(int(px//TILE), int((px+player["w"])//TILE)+1): for y in range(int(py//TILE), int((py+player["h"])//TILE)+1): if solid(get_block(x, y)): return True return False # Player movement def move_player(): keys = pygame.key.get_pressed() player["vx"] = 0 if keys[pygame.K_a]: player["vx"] = -4 if keys[pygame.K_d]: player["vx"] = 4 if keys[pygame.K_SPACE] and player["on_ground"]: player["vy"] = -10 player["vy"] += GRAVITY # X movement player["x"] += player["vx"] if rect_collide(player["x"], player["y"]): player["x"] -= player["vx"] # Y movement player["y"] += player["vy"] if rect_collide(player["x"], player["y"]): player["y"] -= player["vy"] player["vy"] = 0 player["on_ground"] = True else: player["on_ground"] = False # Zombies def spawn_zombie(): zombies.append({"x": random.randint(0, WORLD_WIDTH*TILE), "y": 0, "hp": 10}) def update_zombies(): for z in zombies: if z["x"] < player["x"]: z["x"] += 1 else: z["x"] -= 1 if abs(z["x"] - player["x"]) < 30: player["hp"] -= 0.05 # Items def drop_item(x, y, block): items.append({"x": x, "y": y, "type": block}) def update_items(): for it in items: if abs(it["x"] - player["x"]) < 30 and abs(it["y"] - player["y"]) < 30: player["inventory"][it["type"]] += 1 items.remove(it) # Dragon def spawn_dragon(): dragon["alive"] = True def update_dragon(): if dragon["alive"]: dragon["x"] += math.sin(pygame.time.get_ticks()/400)*5 dragon["y"] += math.cos(pygame.time.get_ticks()/400)*3 if abs(player["x"] - dragon["x"]) < 120: player["hp"] -= 0.2 # Camera def update_camera(): global camera_x, camera_y camera_x = player["x"] - WIDTH//2 camera_y = player["y"] - HEIGHT//2 # Draw def draw_world(): for x in range(WORLD_WIDTH): for y in range(WORLD_HEIGHT): block = world[x][y] if block != "air": pygame.draw.rect(screen, COLORS[block], ( x*TILE - camera_x, y*TILE - camera_y, TILE, TILE )) def draw_player(): pygame.draw.rect(screen, (255,0,0), ( player["x"] - camera_x, player["y"] - camera_y, player["w"], player["h"] )) def draw_zombies(): for z in zombies: pygame.draw.rect(screen, (0,200,0), ( z["x"] - camera_x, z["y"] - camera_y, 30, 40 )) def draw_items(): for it in items: pygame.draw.circle(screen, (255,255,0), ( int(it["x"] - camera_x), int(it["y"] - camera_y) ), 5) def draw_dragon(): if dragon["alive"]: pygame.draw.circle(screen, (200,0,200), ( int(dragon["x"] - camera_x), int(dragon["y"] - camera_y) ), 40) # Game loop running = True spawn_timer = 0 while running: screen.fill((0,0,0)) for event in pygame.event.get(): if event.type == pygame.QUIT: running = False if event.type == pygame.MOUSEBUTTONDOWN: mx, my = pygame.mouse.get_pos() wx = int((mx + camera_x)//TILE) wy = int((my + camera_y)//TILE) if 0 <= wx < WORLD_WIDTH and 0 <= wy < WORLD_HEIGHT: if event.button == 1: block = world[wx][wy] world[wx][wy] = "air" if block != "air": drop_item(wx*TILE, wy*TILE, block) elif event.button == 3: if player["inventory"]["dirt"] > 0: world[wx][wy] = "dirt" player["inventory"]["dirt"] -= 1 move_player() update_camera() update_zombies() update_items() update_dragon() spawn_timer += 1 if spawn_timer > 180: spawn_zombie() spawn_timer = 0 if player["x"] > 2500 and not dragon["alive"]: spawn_dragon() draw_world() draw_player() draw_zombies() draw_items() draw_dragon() pygame.display.flip() clock.tick(60) pygame.quit()
